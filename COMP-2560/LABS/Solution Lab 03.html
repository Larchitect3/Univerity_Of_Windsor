#!/bin/bash

if [ $# == 0 ]; then
    echo "Usage: $0 [-t targetDir]  [-c countFile] <ext-list>"
    exit
fi

targetDir=./             # default target directory
countFile=/dev/stdout    # default result file is screen (standard output)

while getopts ":t:c:" options; do  # take care of possible options
    case "${options}" in 
    t)                   
      targetDir=${OPTARG}     
      ;;
    c)  
      countFile=${OPTARG}   
      ;;
    *)
      echo "Error: -${OPTARG} is an invalid option"
      exit                     
      ;;
  esac
done

shift $((OPTIND-1))   # advance to skip options

echo target Directory is $targetDir  and result file is $countFile

echo -n > $countFile     # in case file exist, get rid of old contents

for ext in $* ; do      # for each extension save the number of file
    echo $ext `ls $targetDir/*.$ext | wc -w` >> $countFile
done
exit 0
